    • rm -rf Premium
    • pkg update && pkg upgrade -y
    • pkg install git
    • pkg install python
    • pip install requests
    • pip install bs4
    • pip install rich
    • pip install pycryptodomex
    • pkg install clang python libffi openssl libsodium
